# Connect remote git repo to local machine:
cd dotfiles-git/
git init
git remote add origin https://github.com/kietoparao/dotfiles.git
git pull https://github.com/kietoparao/dotfiles.git master
git reset

# Set tracking from the remote repo:
git branch -u origin/master

# Push local changes to remote after creating new folders/files on local:
git add .
git commit -a   # OR git commit -m 'custom message'
git push origin master

# Pull changes from the remote repo to the local machine:
git pull https://github.com/kietoparao/dotfiles.git master

# Check repository status
git status
